char* gets(char* dest)
{
    int c=getchar();
    char* p=dest;
    while(c!=EOF&&c!='\n')
    {
        *p++=c;//priority:++(x) > *x > (x)++,*p++ means(*p)++
        c=getchar();
    }
    *p='\0';

}
